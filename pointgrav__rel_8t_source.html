4 subroutine pointgrav(qdt,ixI^L,ixO^L,iw^LIM,qtC,wCT,qt,w,x)
5 
6 !
7 ! Calculate velocity change due to point source gravity
8 ! As it stands now, the points source has to be in the same
9 ! 'dimensionality' as the grid. E.g. It is not possible to
10 ! have an xy grid, with the point source localized in the
11 ! z-direction.
12 ! On the other hand, it IS possible to have the point source
13 ! in a location that is not part of the grid. This will in
14 ! fact often be the case in spherical symmetries with the
15 ! gravitational source at r=0
16 !
17 ! the scaled value for the central mass: ptmass has to be set in
18 ! the parfile, together with the coordinates x1ptms,x2ptms,x3ptms
19 !
20 
21 
22 use mod_global_parameters
23 
24 
25 integer, intent(in) :: ixI^L, ixO^L, iw^LIM
26 double precision, intent(in) :: qdt, qtC, qt, x(ixI^S,1:ndim)
27 double precision, intent(inout) :: wCT(ixI^S,1:nw), w(ixI^S,1:nw)
28 
29 double precision :: r2i(ixG^T,1:ndim)
30 !-----------------------------------------------------------------------------
31 
32 if (ptmass == zero) return
33 
34 
35 call getgvector(ixi^l,ixo^l,x,r2i)
36 
37 w(ixo^s,e_) = w(ixo^s,e_) &
38  - qdt *ptmass*wct(ixo^s,s1_)*r2i(ixo^s,1)
39 {^nooned
40 w(ixo^s,e_) = w(ixo^s,e_) &
41  - qdt *ptmass*wct(ixo^s,s2_)*r2i(ixo^s,2)
42 }
43 {^ifthreed
44 w(ixo^s,e_) = w(ixo^s,e_) &
45  - qdt *ptmass*wct(ixo^s,s3_)*r2i(ixo^s,3)
46 }
47 
48 !
49 ! Update momentum
50 !
51 
52 w(ixo^s,s1_) = w(ixo^s,s1_) &
53  - qdt *ptmass*wct(ixo^s,rho_)*r2i(ixo^s,1)
54 {^nooned
55 w(ixo^s,s2_) = w(ixo^s,s2_) &
56  - qdt *ptmass*wct(ixo^s,rho_)*r2i(ixo^s,2)
57 }
58 {^ifthreed
59 w(ixo^s,s3_) = w(ixo^s,s3_) &
60  - qdt *ptmass*wct(ixo^s,rho_)*r2i(ixo^s,3)
61 }
62 
63 return

109 subroutine getgvector(ixI^L,ixO^L,x,r2i)

111 !
112 ! returns the vector components of the 1/r^2 vector:
113 ! dx1/r^3,
114 ! dx2/r^3,
115 ! dx3/r^3,
116 ! between a pre-determined pointsource and a point in the grid.
117 !
118 
119 
120 use mod_global_parameters
121 
122 integer, intent(in) :: ixO^L,ixI^L
123 double precision, intent(in) :: x(ixI^S,1:ndim)
124 double precision, intent(out) :: r2i(ixG^T,1:ndim)
125 
126 double precision :: rstptms, rctptms
127 
128 double precision, allocatable :: r2(:^D&)
129 {^NOONED
130 double precision, allocatable :: rst(:^D&), rct(:^D&)
131 }
132 !-----------------------------------------------------------------------------
133 
134 allocate(r2(ixo^s))
135 
136 
137 select case (coordinate)
138 case(cartesian,cartesian_stretched)
139 {^ifoned
140  r2(ixo^s) = (x(ixo^s,1)-x1ptms)**2
141 }
142 {^iftwod
143  r2(ixo^s) = (x(ixo^s,1)-x1ptms)**2 &
144  + (x(ixo^s,2)-x2ptms)**2
145 }
146 {^ifthreed
147  r2(ixo^s) = (x(ixo^s,1)-x1ptms)**2 &
148  + (x(ixo^s,2)-x2ptms)**2 &
149  + (x(ixo^s,3)-x3ptms)**2
150 }
151 
152 
153  r2i(ixo^s,1) = (x(ixo^s,1)-x1ptms) &
154  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
155 {^nooned
156  r2i(ixo^s,2) = (x(ixo^s,2)-x2ptms) &
157  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
158 }
159 {^ifthreed
160  r2i(ixo^s,3) = (x(ixo^s,3)-x3ptms) &
161  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
162 }
163 
164 case(cylindrical)
165 {^ifoned
166  r2(ixo^s) = (x(ixo^s,r_)-x1ptms)**2
167  r2i(ixo^s,1) = one / (r2(ixo^s) + smalldouble)
168 }
169 {^iftwod
170  if(^phi == 2) then
171  r2(ixo^s) = (x(ixo^s,r_)-x1ptms)**2 &
172  + two*x(ixo^s,r_)*x1ptms &
173  * (one -cos(x(ixo^s,2)-x2ptms))
174 
175  r2i(ixo^s,1) = (x(ixo^s,r_)-x1ptms*cos(x(ixo^s,2)-x2ptms)) &
176  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
177 
178  r2i(ixo^s,2) = x1ptms * sin(x(ixo^s,2)-x2ptms) &
179  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
180  else
181  r2(ixo^s) = (x(ixo^s,r_)-x1ptms)**2 &
182  + (x(ixo^s,2)-x2ptms)**2
183 
184  r2i(ixo^s,1) = (x(ixo^s,r_)-x1ptms) &
185  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
186  r2i(ixo^s,2) = (x(ixo^s,2)-x2ptms) &
187  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
188  endif
189 }
190 {^ifthreed
191  r2(ixo^s) = (x(ixo^s,r_)-x1ptms)**2 &
192  + (x(ixo^s,z_)-x2ptms)**2 &
193  + two*x(ixo^s,r_)*x1ptms &
194  * (one -cos(x(ixo^s,phi_)-x3ptms))
195 
196  r2i(ixo^s,1) = (x(ixo^s,r_)-x1ptms*cos(x(ixo^s,phi_)-x3ptms)) &
197  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
198  r2i(ixo^s,2) = (x(ixo^s,z_)-x2ptms) &
199  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
200  r2i(ixo^s,3) = x1ptms * sin(x(ixo^s,phi_)-x3ptms) &
201  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
202 }
203 
204 
205 case(spherical)
206  rstptms = x1ptms * sin(x2ptms)
207  rctptms = x1ptms * cos(x2ptms)
208 
209 {^ifoned
210  r2(ixo^s) = (x(ixo^s,r_)-x1ptms)**2
211  r2i(ixo^s,1) = one / (r2(ixo^s) + smalldouble)
212 }
213 {^iftwod
214  allocate(rct(ixo^s))
215  allocate(rst(ixo^s))
216 
217  rct(ixo^s) = x(ixo^s,r_)*cos(x(ixo^s,2))
218  rst(ixo^s) = x(ixo^s,r_)*sin(x(ixo^s,2))
219 
220  r2(ixo^s) = (rct(ixo^s)-rctptms)**2 &
221  + (rst(ixo^s)-rstptms)**2
222 
223 
224  r2i(ixo^s,1) = ((rct(ixo^s)-rctptms) * cos(x(ixo^s,2)) &
225  + (rst(ixo^s)-rstptms) * sin(x(ixo^s,2))) &
226  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
227  r2i(ixo^s,2) = (-(rct(ixo^s)-rctptms) * sin(x(ixo^s,2)) &
228  + (rst(ixo^s) - rstptms) * cos(x(ixo^s,2))) &
229  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
230 
231  deallocate(rst)
232  deallocate(rct)
233 }
234 {^ifthreed
235  allocate(rct(ixo^s))
236  allocate(rst(ixo^s))
237 
238  rct(ixo^s) = x(ixo^s,r_)*cos(x(ixo^s,2))
239  rst(ixo^s) = x(ixo^s,r_)*sin(x(ixo^s,2))
240 
241  r2(ixo^s) = (rct(ixo^s)-rctptms)**2 &
242  + (rst(ixo^s)-rstptms)**2 &
243  + two * rst(ixo^s) * rstptms &
244  * (one-cos(x(ixo^s,phi_)-x3ptms))
245 
246  r2i(ixo^s,1) = ((rct(ixo^s)-rctptms) * cos(x(ixo^s,2)) &
247  + (rst(ixo^s)-rstptms*cos(x(ixo^s,phi_)-x3ptms)) &
248  * sin(x(ixo^s,2))) &
249  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
250  r2i(ixo^s,2) = (-(rct(ixo^s)-rctptms) * sin(x(ixo^s,2)) &
251  + (rst(ixo^s)-rstptms*cos(x(ixo^s,phi_)-x3ptms)) &
252  * cos(x(ixo^s,2))) &
253  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
254  r2i(ixo^s,3) = rstptms * sin(x(ixo^s,phi_)-x3ptms) &
255  / (r2(ixo^s) * sqrt(r2(ixo^s)) + smalldouble)
256 
257 
258  deallocate(rct)
259  deallocate(rst)
260 }
261  end select
262 
263 deallocate(r2)
264 
265 return
